  git remote -v

  # 添加fork源仓库的地址
  git remote add upstream ...

 #查看本地仓库关联的远程地址

  # 获取fork源仓库的更新， 实现本地同步
  git fetch upstream
  git checkout master
  git merge updsteam/master

  # 提交到fork仓库副本， 实现副本同步
  git push origin master
